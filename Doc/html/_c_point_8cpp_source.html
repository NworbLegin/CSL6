383 COORD_TYPE CPoint::theta() {
384  if (x == 0.0) {
385  if (y >= 0.0) {
386  return CSL_PI * 0.5;
387  }
388  else {
389  return CSL_PI * 1.5;
390  }
391  }
392  COORD_TYPE tan = atanf(y / x);
393  if (x < 0)
394  return (tan + CSL_PI);
395  else if (tan < 0.0)
396  return (tan + CSL_TWOPI);
397  else
398  return (tan);
399 }
